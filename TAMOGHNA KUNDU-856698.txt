create procedure proc3(@empno int, @empname varchar(50),@deptno int,@salary int)
as
begin
insert into employee
values(@empno,@empname,@deptno,@salary)
end

create procedure proc2(@empno int, @empname varchar(50),@deptno int,@salary int)
as
begin
update employee set @empname=@empname,deptno=@deptno,salary=@salary where empno=@empno
end

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data.OleDb;
using System.Data.SqlClient;
using System.Data;

namespace WindowsFormsApp1
{
    class DL
    {
        SqlConnection con = new SqlConnection(@"Data Source=DESKTOP-8KF9DJ1\SQLEXPRESS;Initial Catalog=cognizant;Integrated Security=True");
        public void register(Employee e)
        {
            con.Open();
            SqlCommand cmd = new SqlCommand();
            cmd.Connection = con;
            cmd.CommandType = CommandType.StoredProcedure;
            cmd.CommandText = "proc3";

            cmd.Parameters.Add(new SqlParameter("@empno", SqlDbType.Int));
            cmd.Parameters["@empno"].Value = e.EmpNo;

            cmd.Parameters.Add(new SqlParameter("@empname", SqlDbType.VarChar));
            cmd.Parameters["@empname"].Value = e.EmpName;

            cmd.Parameters.Add(new SqlParameter("@deptno", SqlDbType.Int));
            cmd.Parameters["@deptno"].Value = e.DeptNo;

            cmd.Parameters.Add(new SqlParameter("@salary", SqlDbType.Int));
            cmd.Parameters["@salary"].Value = e.Salary;

            cmd.ExecuteNonQuery();
            con.Close();
        }
        public void update(Employee e)
        {
            con.Open();
            SqlCommand cmd = new SqlCommand();
            cmd.Connection = con;
            cmd.CommandType = CommandType.StoredProcedure;
            cmd.CommandText = "proc2";

            cmd.Parameters.Add(new SqlParameter("@empno", SqlDbType.Int));
            cmd.Parameters["@empno"].Value = e.EmpNo;

            cmd.Parameters.Add(new SqlParameter("@empname", SqlDbType.VarChar));
            cmd.Parameters["@empname"].Value = e.EmpName;

            cmd.Parameters.Add(new SqlParameter("@deptno", SqlDbType.Int));
            cmd.Parameters["@deptno"].Value = e.DeptNo;

            cmd.Parameters.Add(new SqlParameter("@salary", SqlDbType.Int));
            cmd.Parameters["@salary"].Value = e.Salary;

            cmd.ExecuteNonQuery();
            con.Close();
        }
    }
}
